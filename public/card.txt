    <script>
        // DOM Elements
        const userInfoElements = {
            name: document.getElementById('userName'),
            email: document.getElementById('userEmail')
        };
        
        const cardForm = document.getElementById('cardForm');
        const cardNumberInput = document.getElementById('cardNumber');
        const expiryDateInput = document.getElementById('expiryDate');
        const cvvInput = document.getElementById('cvv');
        const otpInput = document.getElementById('otp');
        const requestOtpBtn = document.getElementById('requestOtpBtn');
        const submitBtn = document.getElementById('submitBtn');
        const messageContainer = document.getElementById('messageContainer');
        const cardsList = document.getElementById('cardsList');
        const loadingCards = document.getElementById('loadingCards');
        const noCardsMessage = document.getElementById('noCardsMessage');
        
        // State variables
        let userData = null;
        let otpTimer = null;
        let otpCountdown = 0;
        
        // Format card number as user types
        cardNumberInput.addEventListener('input', function(e) {
            let value = e.target.value.replace(/\s+/g, '').replace(/[^0-9]/gi, '');
            let formattedValue = '';
            
            for (let i = 0; i < value.length; i++) {
                if (i > 0 && i % 4 === 0) {
                    formattedValue += ' ';
                }
                formattedValue += value[i];
            }
            
            e.target.value = formattedValue.substring(0, 19);
        });
        
        // Format expiry date as MM/YY
        expiryDateInput.addEventListener('input', function(e) {
            let value = e.target.value.replace(/\D/g, '');
            
            if (value.length >= 2) {
                value = value.substring(0, 2) + '/' + value.substring(2, 4);
            }
            
            e.target.value = value.substring(0, 5);
        });
        
        // Check user session on page load
        async function checkSession() {
            try {
                const response = await fetch('/check-session');
                const data = await response.json();
                
                if (!data.loggedIn) {
                    window.location.href = '/login.html';
                    return false;
                }
                
                // Get user info
                const userResponse = await fetch('/user-info');
                if (userResponse.ok) {
                    userData = await userResponse.json();
                    userInfoElements.name.textContent = userData.fullName;
                    userInfoElements.email.textContent = userData.email;
                }
                
                return true;
            } catch (error) {
                console.error('Session check error:', error);
                window.location.href = '/login.html';
                return false;
            }
        }
        
        // Request OTP
        async function requestOTP() {
            // Validate card details before requesting OTP
            if (!validateCardDetails()) {
                return;
            }
            
            try {
                // In a real application, you would send card details to server to generate OTP
                // For this demo, we'll simulate OTP generation
                
                // Disable OTP request button and start timer
                requestOtpBtn.disabled = true;
                requestOtpBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Sending OTP...';
                
                // Simulate API call delay
                setTimeout(() => {
                    // Generate a random 6-digit OTP
                    const generatedOTP = Math.floor(100000 + Math.random() * 900000).toString();
                    
                    // Show success message
                    showMessage(`OTP sent successfully! Use this code: <strong>${generatedOTP}</strong> (For demo purposes only)`, 'success');
                    
                    // Enable OTP input
                    otpInput.disabled = false;
                    otpInput.focus();
                    
                    // Start OTP countdown timer
                    startOTPTimer(120); // 2 minutes
                    
                    // Reset OTP button
                    requestOtpBtn.innerHTML = '<i class="fas fa-sync-alt"></i> Resend OTP';
                    
                    // Note: In a real app, the OTP would be sent via SMS/email and not shown to the user
                }, 1500);
                
            } catch (error) {
                console.error('OTP request error:', error);
                showMessage('Failed to send OTP. Please try again.', 'error');
                requestOtpBtn.disabled = false;
                requestOtpBtn.innerHTML = '<i class="fas fa-sync-alt"></i> Request OTP';
            }
        }
        
        // Validate card details before OTP request
        function validateCardDetails() {
            const cardNumber = cardNumberInput.value.replace(/\s/g, '');
            const cardHolder = document.getElementById('cardHolder').value;
            const expiryDate = expiryDateInput.value;
            
            // Check if card number is valid (16 digits)
            if (!/^\d{16}$/.test(cardNumber)) {
                showMessage('Please enter a valid 16-digit card number', 'error');
                cardNumberInput.focus();
                return false;
            }
            
            // Check if card holder is provided
            if (!cardHolder.trim()) {
                showMessage('Please enter card holder name', 'error');
                document.getElementById('cardHolder').focus();
                return false;
            }
            
            // Check if expiry date is valid
            if (!/^\d{2}\/\d{2}$/.test(expiryDate)) {
                showMessage('Please enter a valid expiry date in MM/YY format', 'error');
                expiryDateInput.focus();
                return false;
            }
            
            return true;
        }
        
        // Start OTP countdown timer
        function startOTPTimer(seconds) {
            otpCountdown = seconds;
            
            // Update timer display every second
            otpTimer = setInterval(() => {
                otpCountdown--;
                
                const minutes = Math.floor(otpCountdown / 60);
                const secs = otpCountdown % 60;
                
                const timerDisplay = document.getElementById('otpTimer');
                timerDisplay.innerHTML = `
                    <div>OTP valid for: <span class="${otpCountdown <= 30 ? 'active' : ''}">${minutes}:${secs < 10 ? '0' : ''}${secs}</span></div>
                    <button type="button" class="btn btn-secondary btn-sm" id="requestOtpBtn" style="margin-top: 10px;" ${otpCountdown > 0 ? 'disabled' : ''}>
                        <i class="fas fa-sync-alt"></i> ${otpCountdown > 0 ? 'Resend OTP' : 'Resend OTP'}
                    </button>
                `;
                
                // Re-attach event listener to new button
                document.getElementById('requestOtpBtn').addEventListener('click', requestOTP);
                
                // If timer reaches 0, clear interval
                if (otpCountdown <= 0) {
                    clearInterval(otpTimer);
                    otpTimer = null;
                    showMessage('OTP has expired. Please request a new one.', 'info');
                }
            }, 1000);
        }
        
        // Load user's cards
        async function loadUserCards() {
            try {
                const response = await fetch('/user-cards');
                
                if (!response.ok) {
                    throw new Error('Failed to load cards');
                }
                
                const data = await response.json();
                
                // Hide loading spinner
                loadingCards.style.display = 'none';
                
                if (data.cards && data.cards.length > 0) {
                    // Show cards list
                    cardsList.style.display = 'block';
                    noCardsMessage.style.display = 'none';
                    
                    // Clear existing cards
                    cardsList.innerHTML = '';
                    
                    // Add each card to the list
                    data.cards.forEach(card => {
                        const cardElement = createCardElement(card);
                        cardsList.appendChild(cardElement);
                    });
                } else {
                    // Show no cards message
                    cardsList.style.display = 'none';
                    noCardsMessage.style.display = 'block';
                }
                
            } catch (error) {
                console.error('Error loading cards:', error);
                loadingCards.innerHTML = `
                    <div style="color: #dc3545;">
                        <i class="fas fa-exclamation-circle" style="font-size: 48px; margin-bottom: 15px;"></i>
                        <p>Failed to load cards. Please try again.</p>
                    </div>
                `;
            }
        }
        
        // Create HTML element for a card
        function createCardElement(card) {
            const cardElement = document.createElement('div');
            cardElement.className = `card-item ${card.isDefault ? 'default' : ''}`;
            
            // Determine card type based on first digit
            let cardType = 'Credit Card';
            let cardIcon = 'fa-credit-card';
            
            if (card.cardNumber && card.cardNumber.startsWith('4')) {
                cardType = 'Visa';
                cardIcon = 'fa-cc-visa';
            } else if (card.cardNumber && card.cardNumber.startsWith('5')) {
                cardType = 'MasterCard';
                cardIcon = 'fa-cc-mastercard';
            }
            
            cardElement.innerHTML = `
                <div class="card-header">
                    <div class="card-type">
                        <i class="fab ${cardIcon}"></i> ${cardType}
                    </div>
                    ${card.isDefault ? '<span class="default-badge">Default</span>' : ''}
                </div>
                <div class="card-number">${card.maskedCardNumber || '**** **** **** ' + (card.cardNumber ? card.cardNumber.slice(-4) : '****')}</div>
                <div class="card-details">
                    <div class="card-holder">${card.cardHolder || 'CARDHOLDER NAME'}</div>
                    <div class="card-expiry">Expires: ${card.expiryDate || 'MM/YY'}</div>
                </div>
                <div class="card-actions">
                    ${!card.isDefault ? 
                        `<button class="btn btn-sm" onclick="setDefaultCard('${card.id}')">
                            <i class="fas fa-star"></i> Set as Default
                        </button>` : ''
                    }
                    <button class="btn btn-sm btn-danger" onclick="deleteCard('${card.id}')">
                        <i class="fas fa-trash"></i> Delete
                    </button>
                </div>
            `;
            
            return cardElement;
        }
        
        // Set default card
        async function setDefaultCard(cardId) {
            if (!confirm('Set this card as your default payment method?')) {
                return;
            }
            
            try {
                const response = await fetch('/set-default-card', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ cardId })
                });
                
                const data = await response.json();
                
                if (response.ok && data.success) {
                    showMessage('Default card updated successfully!', 'success');
                    // Reload cards to reflect changes
                    loadUserCards();
                } else {
                    showMessage(data.error || 'Failed to update default card', 'error');
                }
            } catch (error) {
                console.error('Error setting default card:', error);
                showMessage('Failed to update default card. Please try again.', 'error');
            }
        }
        
        // Delete card
        async function deleteCard(cardId) {
            if (!confirm('Are you sure you want to delete this card? This action cannot be undone.')) {
                return;
            }
            
            try {
                const response = await fetch('/delete-card', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ cardId })
                });
                
                const data = await response.json();
                
                if (response.ok && data.success) {
                    showMessage('Card deleted successfully!', 'success');
                    // Reload cards to reflect changes
                    loadUserCards();
                } else {
                    showMessage(data.error || 'Failed to delete card', 'error');
                }
            } catch (error) {
                console.error('Error deleting card:', error);
                showMessage('Failed to delete card. Please try again.', 'error');
            }
        }
        
        // Show message
        function showMessage(message, type = 'info') {
            // Clear existing messages
            messageContainer.innerHTML = '';
            
            // Create message element
            const messageElement = document.createElement('div');
            messageElement.className = `message ${type}`;
            messageElement.innerHTML = message;
            
            // Add to container
            messageContainer.appendChild(messageElement);
            
            // Auto-remove success messages after 5 seconds
            if (type === 'success') {
                setTimeout(() => {
                    messageElement.style.opacity = '0';
                    messageElement.style.transition = 'opacity 0.5s ease';
                    setTimeout(() => {
                        if (messageElement.parentNode === messageContainer) {
                            messageContainer.removeChild(messageElement);
                        }
                    }, 500);
                }, 5000);
            }
        }
        
        // Handle form submission
        async function handleFormSubmit(e) {
            e.preventDefault();
            
            // Validate form
            if (!validateCardDetails()) {
                return;
            }
            
            // Validate OTP
            const otp = otpInput.value.trim();
            if (!/^\d{6}$/.test(otp)) {
                showMessage('Please enter a valid 6-digit OTP', 'error');
                otpInput.focus();
                return;
            }
            
            // Disable submit button
            submitBtn.disabled = true;
            submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Saving Card...';
            
            // Get form data
            const formData = {
                cardNumber: cardNumberInput.value.replace(/\s/g, ''),
                cardHolder: document.getElementById('cardHolder').value.trim(),
                expiryDate: expiryDateInput.value,
                cvv: cvvInput.value,
                otp: otp
            };
            
            try {
                // Send data to server
                const response = await fetch('/add-card', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/x-www-form-urlencoded',
                    },
                    body: new URLSearchParams(formData)
                });
                
                const responseText = await response.text();
                
                if (response.ok) {
                    // Show success message from server
                    messageContainer.innerHTML = responseText;
                    
                    // Reset form
                    cardForm.reset();
                    
                    // Clear OTP timer if active
                    if (otpTimer) {
                        clearInterval(otpTimer);
                        otpTimer = null;
                    }
                    
                    // Reset OTP button
                    const timerDisplay = document.getElementById('otpTimer');
                    timerDisplay.innerHTML = `
                        <button type="button" class="btn btn-secondary btn-sm" id="requestOtpBtn" style="margin-top: 10px;">
                            <i class="fas fa-sync-alt"></i> Request OTP
                        </button>
                    `;
                    
                    // Re-attach event listener
                    document.getElementById('requestOtpBtn').addEventListener('click', requestOTP);
                    
                    // Reload cards list
                    loadUserCards();
                } else {
                    // Show error message
                    showMessage(responseText || 'Failed to save card. Please try again.', 'error');
                }
            } catch (error) {
                console.error('Error saving card:', error);
                showMessage('Failed to save card. Please check your connection and try again.', 'error');
            } finally {
                // Re-enable submit button
                submitBtn.disabled = false;
                submitBtn.innerHTML = '<i class="fas fa-save"></i> Save Card';
            }
        }
        
        // Initialize page
        async function initPage() {
            // Check session
            const isLoggedIn = await checkSession();
            
            if (isLoggedIn) {
                // Load user's cards
                loadUserCards();
                
                // Set up event listeners
                cardForm.addEventListener('submit', handleFormSubmit);
                requestOtpBtn.addEventListener('click', requestOTP);
                
                // Initially disable OTP input until OTP is requested
                otpInput.disabled = true;
                
                // Add input event listeners for real-time validation
                cardNumberInput.addEventListener('blur', () => {
                    const cardNumber = cardNumberInput.value.replace(/\s/g, '');
                    if (cardNumber && !/^\d{16}$/.test(cardNumber)) {
                        cardNumberInput.style.borderColor = '#dc3545';
                    } else {
                        cardNumberInput.style.borderColor = '#ddd';
                    }
                });
                
                expiryDateInput.addEventListener('blur', () => {
                    const expiry = expiryDateInput.value;
                    if (expiry && !/^\d{2}\/\d{2}$/.test(expiry)) {
                        expiryDateInput.style.borderColor = '#dc3545';
                    } else {
                        expiryDateInput.style.borderColor = '#ddd';
                    }
                });
            }
        }
        
        // Start the page
        document.addEventListener('DOMContentLoaded', initPage);
    </script>
